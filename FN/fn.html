    # 通过子节点寻找父级节点

    /**
 *  @param { json }  Object  data
 *  @param { dom }   String  子节点
 *  @returns [ String ] id 
*/
function filtername(json, dom) {
    var nameArr
    function filterDom (json, dom, nameList) {
        var list = nameList
        for (var index = 0; index < json.length; index++) {
            const item = json[index];
            let childrenList = []
            if (item.children.length) {
                for (let i = 0; i < item.children.length; i++) {
                    const obj = item.children[i];
                    childrenList.push(obj)
                }
            }
            if (item.name === dom) {
                nameArr = [...list, dom]
            } else {
                filterDom(childrenList, dom, [...list, item.name])
            }
        }
    }
    filterDom(json, dom, [])
    return nameArr
}
