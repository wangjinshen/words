    # 前端代码上传到服务器 自动化脚本

    let Client = require('ssh2-sftp-client');
const path = require('path')
const glob = require('glob')
let sftp = new Client();
const localPath = path.join(__dirname, './dist').replace(/\\/g, '/')
const remotePath = '/data'
sftp.connect({
    host: '',
    port: '22',
    username: 'root',
    password: ''
}).then(() => {
    return sftp.rmdir(`${remotePath}/static`, true)
}).then(() => {
    return Promise.all([
        sftp.mkdir(`${remotePath}/static`, true),
        sftp.mkdir(`${remotePath}/static/css`, true),
        sftp.mkdir(`${remotePath}/static/img`, true),
        sftp.mkdir(`${remotePath}/static/js`, true)
    ])
}).then(() => {
    const fileLiat = glob.sync(`${localPath}/**/*.{js,css,png,html}`)
    return Promise.all(
        fileLiat.map(file=>{
            const remoteFile = file.replace(localPath, remotePath+"/static")
            return sftp.fastPut(file,remoteFile)
        })
    )
}).then(() => {
    console.log('成功');
    sftp.end()
}).catch((err) => {
    console.log(err.message, 'catch error');
    sftp.end()
});
